// Menu Init
$('.left-menu-wrapper').ultraMenu({

    //Panel AJAX Settings
    ajaxLoad : true,
    ajaxContainer : '.content-panel',
    hashUpdate: true,

    // Collapse Panel ON/OFF
    collapsePanel: true,

    // Panel Width Settings
    panelWidth: 240,
    collapsePanelWidth: 50,

    // JS based Offcanvas Effect
    offCanvas: true,
    offCanvasSpeed: 220,
    offCanvasClass: 'offcanvas-mode'
});
